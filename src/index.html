<body>

    <div id="root"></div>


    <script type="text/babel">

        // Debugging function to check library availability

        const checkLibraries = () => {

            const libraries = [

                { name: 'React', obj: window.React },

                { name: 'ReactDOM', obj: window.ReactDOM },

                { name: 'Recharts', obj: window.Recharts }

            ];


            const missingLibraries = libraries.filter(lib => !lib.obj);

            

            if (missingLibraries.length > 0) {

                const missingNames = missingLibraries.map(lib => lib.name).join(', ');

                console.error(`Missing libraries: ${missingNames}`);

                return false;

            }

            return true;

        };


        // Main Dashboard Component

        const EpidemiologyDashboard = () => {

            // Check if libraries are loaded

            if (!checkLibraries()) {

                return (

                    <div className="error-message">

                        Error: Required libraries are not loaded. 

                        Please check your internet connection and script includes.

                    </div>

                );

            }


            // Destructure Recharts components safely

            const { 

                LineChart, 

                Line, 

                XAxis, 

                YAxis, 

                CartesianGrid, 

                Tooltip, 

                Legend, 

                ResponsiveContainer 

            } = window.Recharts;


            // State hooks

            const [population, setPopulation] = React.useState(10000);

            const [incidence, setIncidence] = React.useState(500);

            const [recoveryRate, setRecoveryRate] = React.useState(20);


            // Generate time series data

            const generateTimeSeriesData = () => {

                const years = 6;

                let data = [];

                let currentCases = 0;


                for (let year = 0; year < years; year++) {

                    // Add new cases each year

                    currentCases += incidence;


                    // Apply recovery

                    if (year > 0) {

                        const recoveredCases = Math.min(

                            Math.floor(currentCases * (recoveryRate / 100)), 

                            currentCases

                        );

                        currentCases -= recoveredCases;

                    }


                    // Ensure cases don't exceed population

                    currentCases = Math.min(currentCases, population);


                    data.push({

                        name: `Year ${year}`,

                        cases: currentCases,

                        incidence: incidence

                    });

                }


                console.log('Generated Data:', data);

                return data;

            };


            // Generate data

            const chartData = generateTimeSeriesData();


            return (

                <div>

                    <h1>Epidemiology Dashboard</h1>

                    

                    <div>

                        <label>

                            Population: 

                            <input 

                                type="range" 

                                min="1000" 

                                max="100000" 

                                value={population}

                                onChange={(e) => setPopulation(Number(e.target.value))}

                            />

                            {population}

                        </label>

                        

                        <label>

                            New Cases per Year: 

                            <input 

                                type="range" 

                                min="10" 

                                max="5000" 

                                value={incidence}

                                onChange={(e) => setIncidence(Number(e.target.value))}

                            />

                            {incidence}

                        </label>

                        

                        <label>

                            Recovery Rate (%): 

                            <input 

                                type="range" 

                                min="0" 

                                max="100" 

                                value={recoveryRate}

                                onChange={(e) => setRecoveryRate(Number(e.target.value))}

                            />

                            {recoveryRate}%

                        </label>

                    </div>


                    <div className="chart-container">

                        <ResponsiveContainer width="100%" height={400}>

                            <LineChart data={chartData}>

                                <CartesianGrid strokeDasharray="3 3" />

                                <XAxis dataKey="name" />

                                <YAxis />

                                <Tooltip />

                                <Legend />

                                <Line 

                                    type="monotone" 

                                    dataKey="cases" 

                                    stroke="#8884d8" 

                                    name="Total Cases" 

                                />

                                <Line 

                                    type="monotone" 

                                    dataKey="incidence" 

                                    stroke="#82ca9d" 

                                    name="New Cases" 

                                />

                            </LineChart>

                        </ResponsiveContainer>

                    </div>

                </div>

            );

        };


        // Render the application

        const renderApp = () => {

            try {

                // Use ReactDOM.createRoot for React 18

                const root = ReactDOM.createRoot(document.getElementById('root'));

                root.render(<EpidemiologyDashboard />);

            } catch (error) {

                console.error('Rendering error:', error);

                // Fallback to older ReactDOM.render if createRoot fails

                try {

                    ReactDOM.render(<EpidemiologyDashboard />, document.getElementById('root'));

                } catch (fallbackError) {

                    console.error('Fallback rendering error:', fallbackError);

                    // Display error message in the DOM

                    document.getElementById('root').innerHTML = `

                        <div style="color: red; border: 1px solid red; padding: 10px;">

                            <h2>Error Rendering Dashboard</h2>

                            <p>${fallbackError.message}</p>

                        </div>

                    `;

                }

            }

        };


        // Ensure DOM is fully loaded before rendering

        if (document.readyState === 'loading') {

            document.addEventListener('DOMContentLoaded', renderApp);

        } else {

            renderApp();

        }
